[JkvServer] (Main Class)
    |-- Inicia o ServerSocket
    |-- Mantém o ExecutorService (Thread Pool)
    |-- Instancia o DataStore (Singleton)
    |-- Loop infinito: serverSocket.accept() -> submite ClientHandler para o Pool

[ClientHandler] (implements Runnable)
    |-- Responsável por 1 conexão (Socket)
    |-- Lê InputStream (BufferedReader)
    |-- Envia para CommandParser
    |-- Recebe Command
    |-- Executa Command.execute(dataStore)
    |-- Escreve resposta no OutputStream

[DataStore] (O Coração - Shared Resource)
    |-- Map<String, String> storage (HashMap simples - NÃO use ConcurrentHashMap aqui, pois o desafio é o lock manual)
    |-- ReentrantReadWriteLock lock (O desafio de concorrência)
    |-- Métodos: put(), get(), remove(), snapshot()

[CommandParser]
    |-- parse(String line) -> Retorna um objeto Command
    |-- Lança InvalidCommandException

[Interface: Command]
    |-- void execute(DataStore store, PrintWriter out)

    [Concrete Commands]
        |-- SetCommand
        |-- GetCommand
        |-- DelCommand
        |-- SaveCommand (O desafio de I/O)
